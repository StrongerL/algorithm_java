// 单调栈
class Solution {
    public int trap(int[] height) {
        Deque<Integer> stack = new LinkedList<>();
        int ans = 0;
        int n = height.length;
        for (int i = 0; i < n; i++) {
            while (!stack.isEmpty() && height[stack.peek()] < height[i]) {
                int idx = stack.pop();
                if (stack.isEmpty()) break;
                int left = stack.peek();
                ans += (i - left - 1) * (Math.min(height[left], height[i]) - height[idx]);
            }
            stack.push(i);
        }
        return ans;
    }
}
